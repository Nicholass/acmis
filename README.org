=CMS for urban exploration blog engine=

==Setup==
#+begin_src shell
 virtualenv --python=python3 --prompt="ACIS" acis_venv
 . ./acis_venv/bin/activate
 pip install -r ./requirements.txt
 python manage.py migrate
 python manage.py runserver
#+end_src

==Run==
#+begin_src shell
 . ./acis_venv/bin/activate
 python manage.py runserver
#+end_src
